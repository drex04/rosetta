   ```ts
     id: string

     sourceHandle: string
     targetHandle: string

   }
   ```

2. In `src/lib/sparql.ts`, update `generateConstruct` to accept the full `Mapping` (minus `id`) and switch on `kind`:
   - `'direct'` (or undefined): current behavior unchanged.
   - `'template'`: emit CONSTRUCT with `BIND(STR(?raw) AS ?val)` and a `# template: {{templatePattern}}` comment above the BIND.
   - `'constant'`: emit CONSTRUCT with no WHERE clause variable — use `BIND(<value>^^<type> AS ?val)`. Where value is `m.constantValue ?? ''` and type is `m.constantType ?? 'http://www.w3.org/2001/XMLSchema#string'`.
   - `'typecast'`: emit CONSTRUCT with `BIND(STRDT(STR(?raw), <targetDatatype>) AS ?val)` using `m.targetDatatype ?? 'http://www.w3.org/2001/XMLSchema#string'`.
   - `'language'`: emit CONSTRUCT with `BIND(STRLANG(STR(?raw), "${m.languageTag ?? 'en'}") AS ?val)`.
   - `'join'`: emit a stub:
     ```sparql
     PREFIX src: <${srcPrefix}>
     PREFIX tgt: <${tgtPrefix}>
     # JOIN placeholder — requires manual implementation
     # parentSource: ${m.parentSourceId ?? '?'}  parentRef: ${m.parentRef ?? '?'}  childRef: ${m.childRef ?? '?'}
     CONSTRUCT {
       ?target tgt:${tgtProp} ?val .
     }
     WHERE {
       FILTER(false) # Remove FILTER and implement join logic
     }
     ```
   - `'sparql'`: return `m.sparqlConstruct` unchanged (user-managed).
3. Update `generateConstruct`'s parameter type from `Omit<Mapping, 'id' | 'sparqlConstruct' | 'kind'>` to `Omit<Mapping, 'id' | 'sparqlConstruct'>` so callers can pass `kind`.
4. In `MappingPanel.tsx` `handleRegenerate()`: pass `selectedMapping` directly (it now includes `kind`).

**FusedTab component** (in same file):
```tsx
function FusedTab() {
  const { result, jsonLd, loading, error, stale, runFusion, setStale } = useFusionStore()
  const mappings = useMappingStore((s) => s.mappings)
  const sources = useSourcesStore((s) => s.sources)

  // Stale detection: mark fusion stale when sources or mappings change
  useEffect(() => {
    if (result !== null) setStale(true)
    // eslint-disable-next-line react-hooks/exhaustive-deps
  }, [mappings, sources])

  const jsonString = jsonLd !== null ? JSON.stringify(jsonLd, null, 2) : null

  function handleDownload(format: 'json' | 'jsonld') {
    if (jsonString === null) return
    const ext = format === 'jsonld' ? 'jsonld' : 'json'
    const mime = format === 'jsonld' ? 'application/ld+json' : 'application/json'
    downloadBlob(`fused.${ext}`, jsonString, mime)
  }

  return (
    <div className="flex flex-col h-full overflow-hidden">
      {/* Toolbar */}
      <div className="shrink-0 flex items-center justify-between px-3 py-2 border-b border-border bg-muted/20">
        <div className="flex items-center gap-2">
          <span className="text-xs font-semibold uppercase tracking-wide text-muted-foreground">
            Fused Output
          </span>
          {stale && result !== null && (
            <span className="text-[10px] px-1.5 py-0.5 rounded bg-amber-100 text-amber-700">
              stale
            </span>
          )}
        </div>
        <Button
          variant="outline"
          size="sm"
          className="h-6 gap-1 text-xs"
          disabled={loading}
          onClick={() => void runFusion()}
        >
          {loading
            ? <CircleNotchIcon size={12} className="animate-spin" />
            : <PlayIcon size={12} />}
          {loading ? 'Running…' : 'Transform & Fuse'}
        </Button>
      </div>

      {/* Error */}
      {error !== null && (
        <div className="shrink-0 flex items-center gap-2 px-3 py-2 bg-destructive/10 text-destructive text-xs">
          <WarningIcon size={14} /> {error}
        </div>
      )}

      {/* Empty state */}
      {result === null && !loading && error === null && (
        <div className="flex-1 flex items-center justify-center p-6">
          <p className="text-xs text-muted-foreground text-center">
            Click <strong>Transform & Fuse</strong> to execute all source mappings and merge into unified output.
          </p>
        </div>
      )}

      {/* Results */}
      {result !== null && jsonString !== null && (
        <>
          {/* Source summary */}
          <div className="shrink-0 px-3 py-2 border-b border-border bg-muted/10 text-xs text-muted-foreground">
            Fused from: {result.sources.map((s) => `${s.sourceName} (${s.quadCount} triples)`).join(', ')}
            {' · '}{result.totalQuads} total triples
          </div>
          {/* Download buttons */}
          <div className="shrink-0 flex items-center gap-2 px-3 py-1.5 border-b border-border">
            <Button variant="outline" size="sm" className="h-6 gap-1 text-xs"
              onClick={() => handleDownload('json')}>
              <DownloadSimpleIcon size={12} /> .json
            </Button>
            <Button variant="outline" size="sm" className="h-6 gap-1 text-xs"
              onClick={() => handleDownload('jsonld')}>
              <DownloadSimpleIcon size={12} /> .jsonld
            </Button>
          </div>
          {/* JSON preview (read-only CodeMirror) */}
          <JsonPreview content={jsonString} />
        </>
      )}
    </div>
  )
}
```

**JsonPreview** (read-only CodeMirror JSON viewer — same pattern as existing JsonLdView):
```tsx
function JsonPreview({ content }: { content: string }) {
  const containerRef = useRef<HTMLDivElement>(null)
  const viewRef = useRef<EditorView | null>(null)
  // Mount once
  useEffect(() => {
    if (!containerRef.current) return
    const state = EditorState.create({
      doc: content,
      extensions: [basicSetup, lineNumbers(), json(), lightTheme, EditorState.readOnly.of(true)],
    })
    const view = new EditorView({ state, parent: containerRef.current })
    viewRef.current = view
    return () => { view.destroy(); viewRef.current = null }
  }, []) // eslint-disable-line react-hooks/exhaustive-deps
  // Sync content
  useEffect(() => {
    const view = viewRef.current
    if (!view) return
    const cur = view.state.doc.toString()
    if (cur === content) return
    view.dispatch({ changes: { from: 0, to: view.state.doc.length, insert: content } })
  }, [content])
  return <div ref={containerRef} className="flex-1 overflow-hidden" aria-label="Fused JSON output" />
}
```
