2. Create src/store/mappingStore.ts using Zustand:
   ```typescript
   import { create } from 'zustand'
   import { nanoid } from 'nanoid'  // or crypto.randomUUID()
   import type { Mapping } from '@/types/index'

   interface MappingState {
     mappings: Record<string, Mapping[]>  // keyed by sourceId
     selectedMappingId: string | null
     addMapping: (m: Omit<Mapping, 'id'>) => string
     removeMapping: (id: string) => void
     updateMapping: (id: string, patch: Partial<Omit<Mapping, 'id'>>) => void
     getMappingsForSource: (sourceId: string) => Mapping[]
     setSelectedMappingId: (id: string | null) => void
   }

   export const useMappingStore = create<MappingState>((set, get) => ({
     mappings: {},
     selectedMappingId: null,
     addMapping: (m) => {
       const id = crypto.randomUUID()
       set((state) => ({
         mappings: {
           ...state.mappings,
           [m.sourceId]: [...(state.mappings[m.sourceId] ?? []), { ...m, id }],
         },
       }))
       return id
     },
     removeMapping: (id) => {
       set((state) => {
         const next: Record<string, Mapping[]> = {}
         for (const [sid, ms] of Object.entries(state.mappings)) {
           next[sid] = ms.filter((m) => m.id !== id)
         }
         return { mappings: next, selectedMappingId: state.selectedMappingId === id ? null : state.selectedMappingId }
       })
     },
     updateMapping: (id, patch) => {
       set((state) => {
         const next: Record<string, Mapping[]> = {}
         for (const [sid, ms] of Object.entries(state.mappings)) {
           next[sid] = ms.map((m) => m.id === id ? { ...m, ...patch } : m)
         }
         return { mappings: next }
       })
     },
     getMappingsForSource: (sourceId) => get().mappings[sourceId] ?? [],
     setSelectedMappingId: (id) => set({ selectedMappingId: id }),
   }))
   ```

3. Write tests in src/__tests__/mappingStore.test.ts covering:
   - addMapping returns an id and stores mapping under sourceId
   - removeMapping clears selectedMappingId when it matches
   - getMappingsForSource returns [] for unknown sourceId
   - updateMapping patches only the targeted mapping

       const { sourceHandle, targetHandle } = connection
       return (
         sourceHandle?.startsWith('prop_') === true &&
         targetHandle?.startsWith('target_prop_') === true

   - Add `onConnect` callback:
     ```typescript
     const addMapping = useMappingStore((s) => s.addMapping)
     const onConnect = useCallback((connection: Connection) => {
       const { source, sourceHandle, target, targetHandle } = connection
       if (!source || !sourceHandle || !target || !targetHandle) return
       const { nodes: allNodes } = useOntologyStore.getState()  // master nodes
       const { activeSourceId, sources } = useSourcesStore.getState()
       if (!activeSourceId) return

       const activeSource = sources.find((s) => s.id === activeSourceId)
       const sourceNode = activeSource?.schemaNodes.find((n) => n.id === source)
       const targetNode = allNodes.find((n) => n.id === target)
       if (!sourceNode || !targetNode) return

       const propLabel = sourceHandle.replace('prop_', '')
       const targetPropLabel = targetHandle.replace('target_prop_', '')
       const sourceProp = sourceNode.data.properties.find((p) => p.label === propLabel)
       const targetProp = targetNode.data.properties.find((p) => p.label === targetPropLabel)
       if (!sourceProp || !targetProp) return

       addMapping({
         sourceId: activeSourceId,
         sourceClassUri: sourceNode.data.uri,
         sourcePropUri: sourceProp.uri,
         targetClassUri: targetNode.data.uri,
         targetPropUri: targetProp.uri,
         kind: 'direct',
         sparqlConstruct: '',  // generator wired in Plan 04-02
       })
     }, [addMapping])
     ```
   - Pass `isValidConnection` and `onConnect` to `<ReactFlow>`
