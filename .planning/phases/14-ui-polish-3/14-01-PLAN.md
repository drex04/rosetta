    Export `async function parseOntologyFile(file: File): Promise&lt;string&gt;`.

    ```typescript
    import { Parser, Writer, Store } from 'n3';
    import * as jsonld from 'jsonld';

    export async function parseOntologyFile(file: File): Promise&lt;string&gt; {
      const text = await file.text();
      const ext = file.name.split('.').pop()?.toLowerCase() ?? '';

      if (ext === 'ttl') {
        // Parse to validate, then re-serialize to normalize
        return await turtleToTurtle(text);
      }

      if (ext === 'jsonld') {
        const doc: unknown = JSON.parse(text);
        const nquads = await (jsonld as unknown as {
          toRDF: (doc: unknown, opts: object) => Promise&lt;string&gt;;
        }).toRDF(doc, { format: 'application/n-quads' });
        return await nquadsToTurtle(nquads);
      }

      if (ext === 'rdf') {
        return await rdfXmlToTurtle(text);
      }

      throw new Error(`Unsupported format: .${ext}. Use .ttl, .rdf, or .jsonld.`);
    }

    function turtleToTurtle(text: string): Promise&lt;string&gt; {
      return new Promise((resolve, reject) => {
        const store = new Store();
        new Parser().parse(text, (err, quad) => {
          if (err) { reject(new Error(`Invalid Turtle: ${err.message}`)); return; }
          if (quad) { store.add(quad); return; }
          // quad===null means done
          const writer = new Writer({ prefixes: {} });
          store.forEach(q =&gt; writer.addQuad(q));
          writer.end((e, result) =&gt; e ? reject(e) : resolve(result));
        });
      });
    }

    function nquadsToTurtle(nquads: string): Promise&lt;string&gt; {
      return new Promise((resolve, reject) => {
        const store = new Store();
        new Parser({ format: 'N-Quads' }).parse(nquads, (err, quad) => {
          if (err) { reject(err); return; }
          if (quad) { store.add(quad); return; }
          const writer = new Writer({ prefixes: {} });
          store.forEach(q =&gt; writer.addQuad(q));
          writer.end((e, result) =&gt; e ? reject(e) : resolve(result));
        });
      });
    }

    async function rdfXmlToTurtle(text: string): Promise&lt;string&gt; {
      const { RdfXmlParser } = await import('rdfxml-streaming-parser');
      const store = new Store();
      return new Promise((resolve, reject) => {
        const parser = new RdfXmlParser();
        parser.on('data', (quad) =&gt; store.add(quad));
        parser.on('error', (err: Error) =&gt;
          reject(new Error(`Invalid RDF/XML: ${err.message}`)));
        parser.on('end', () =&gt; {
          const writer = new Writer({ prefixes: {} });
          store.forEach(q =&gt; writer.addQuad(q));
          writer.end((e, result) =&gt; e ? reject(e) : resolve(result));
        });
        parser.write(text);
        parser.end();
      });
    }
    ```
