  <files>src/components/layout/RightPanel.tsx</files>
  <action>
    Rewrite RightPanel.tsx width/collapse management:

    1. Add state and mobile detection:
       ```tsx
       const [collapsed, setCollapsed] = useState(false)
       const [width, setWidth] = useState(() => Math.round(window.innerWidth * 0.3))
       const [isMobile, setIsMobile] = useState(() => window.innerWidth < 640)

       useEffect(() => {
         const handler = () => setIsMobile(window.innerWidth < 640)
         window.addEventListener('resize', handler)
         return () => window.removeEventListener('resize', handler)
       }, [])
       ```

    2. Pointer drag handler (desktop only — only attach when !isMobile && !collapsed):
       ```tsx
       function handlePointerDown(e: React.PointerEvent<HTMLDivElement>) {
         e.preventDefault()
         const startX = e.clientX
         const startWidth = width
         function onMove(ev: PointerEvent) {
           const next = Math.min(Math.round(window.innerWidth * 0.6), Math.max(260, startWidth + startX - ev.clientX))
           setWidth(next)
         }
         function onUp() {
           document.removeEventListener('pointermove', onMove)
           document.removeEventListener('pointerup', onUp)
         }
         document.addEventListener('pointermove', onMove)
         document.addEventListener('pointerup', onUp)
       }
       ```

    3. Determine aside style and class based on state:
       - Collapsed: `w-10 shrink-0` (fixed narrow strip), no inline style width
       - Mobile + expanded: `w-full absolute inset-y-0 right-0 z-20` (overlays canvas)
       - Desktop + expanded: `shrink-0`, `style={{ width }}`

    4. Collapsed strip content (rendered instead of Tabs when collapsed):
       ```tsx
       {collapsed ? (
         <div className="flex flex-col items-center justify-start pt-3 h-full gap-2">

             onClick={() => setCollapsed(false)}

             aria-label="Expand panel"
           >
             <CaretLeftIcon size={14} />  {/* points left = expand (panel is on the right) */}
           </button>
           {/* Current tab label rotated */}
           <span className="text-[10px] text-muted-foreground [writing-mode:vertical-rl] rotate-180 select-none mt-1">
             {activeRightTab}
           </span>
         </div>
       ) : (
         <>
           {/* Drag handle — desktop only */}
           {!isMobile && (
             <div
               onPointerDown={handlePointerDown}
               className="absolute left-0 top-0 h-full w-1 cursor-col-resize bg-border hover:bg-primary/40 transition-colors z-10"
               aria-hidden="true"
             />
           )}
           {/* Collapse button inside tab bar area */}
           {/* ... add a small collapse button to the tab bar div ... */}
           {/* Full Tabs content unchanged */}
         </>
       )}
       ```
